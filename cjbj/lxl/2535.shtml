　　五、编写程序（24分）

　　1.main（）

　　{ float x，y；

　　char operator；

　　scanf（“%f，%c，%f”，&x，&operator，&y）

　　switch（operator）

　　{ case `+`：printf（“\n%.2f+%.2f”，x，y，x+y）；break；

　　case `-`：printf（“\n%.2f-%.2f”，x，y，x-y）；break；

　　case `*`：printf（“\n%.2f-%.2f”，x，y，x*y）；break；

　　case `/`：printf（“\n%.2f-%.2f”，x，y，x/y）；break；

　　case `/`： if（y==0）

　　{printf（“error！”）；break；}

　　printf（“\n%.2f-%.2f”，x，y，x/y）；break；

　　default：printf（“operator is invalid”）；}}

　　2. #include “math.h”

　　main（）

　　{ int m，k，i，n；

　　for（m=3；m<=100；m+=2）

　　{  k=sqrt（m）；

　　for（i=2；i<=k；i++）

　　if（m%i==0） break；

　　if（i>=k+1）

　　{printf（“%d”，m）；

　　n++；}

　　}}

　　3. #include “stdio.h”

　　#define  ROW   3

　　#define  COL   4

　　main（）

　　{int i，j，a[ROW][COL]，  a[ROW][COL]；

　　for（i=0；i<=ROW；i++）

　　for（j=0；j<=COL；j++）

　　scanf（“%d”，&a[i][j]）；

　　for（i=0；i<=ROW；i++）

　　for（j=0；j<=COL；j++）

　　b[j][i]=a[i][j]；

　　for（i=0；i<=ROW；i++）

　　for（j=0；j<=COL；j++）

　　printf（“%5d”，b[i][j]）；}}
