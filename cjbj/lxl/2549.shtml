　　六、编程题

　　1.main（）

　　{    int  a[10]；

　　int   m，n，t；

　　for （m=0；m<10；m++）

　　scanf（“%d”，&a[m]）；

　　for（m=0；m<9；m++）

　　for（n=0；n<9；n++）

　　if（a[n]>a[n+1]）

　　{   t=a[n]；  a[n]=a[n+1]；   a[n+1]=t；}

　　for （m=0；m<10；m++）

　　printf（“%d”，a[m]）；

　　}

　　2.

　　main（）

　　{   float   x，y；

　　scanf（“%f”，&x）；

　　if（x>6）

　　y=（x+2）/（3*x+1）；

　　else

　　y=x*x+3*x+1；

　　printf（“%f”，y）；

　　}

　　3.main（）

　　{   FILE   *fp1 ， *fp2；

　　fp1=fopen（“file1.dat”， “r”）；

　　fp2=fopen（“file1.dat”， “w”）；

　　ch=fgetc（fp1）；

　　while（ch！=EOF）

　　{

　　if（ch>=‘a’ && ch<=‘z’）

　　}bsp；   }
