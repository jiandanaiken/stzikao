　　1.#include “stdio.h”

　　main（）

　　{ int term[10]；

　　int i，max，min；

　　printf（“请输入10个整数：”）；

　　for（i=0；i<10；i++）

　　scanf（“%d”，&term[i]）；

　　（max=term[0]；

　　min=term[0]；

　　for（i=0；i<10；i++）

　　{if（term[i]<min）min=term[i]；

　　if（term[i]>max）max=term[i]；}

　　printf（“max=%d，min=%d”，max，min）；}

　　2. main（）

　　int j，k，x；

　　int a[][3]={9，7，5，3，1，2，4，6，8}；

　　for（j=0；j<3；j++）

　　for（k=j+1，k<3；k++）

　　{x=a[j][k]；a[j][k]=a[k][j]；a[k][j]=x；}

　　for（j=0；j<3；j++）

　　{for（k=0，k<3；k++）

　　printf（“%4d”，a[j][k]）；}}

　　return（m*n/gcd（m，n））；}
